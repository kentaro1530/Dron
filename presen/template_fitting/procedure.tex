\begin{frame}{Template fitting procedure}
  \begin{enumerate}
  \item Fitting to evaluate background. \\
    Fitting invariant masses of $n_{forward} \pi^+$, $n_{forward} \pi^-$, and $\pi^+ \pi^-$\\
    for $\Sigma^+_{forward}$, $\Sigma^-_{forward}$, and $K^0$ peaks, respectively. \\
    $\Rightarrow$ Event Sample $d(K^-, n \pi^-\pi^+)"n"$
  \item Fitting to separate $\pi^-\Sigma^+$ and $\pi^+\Sigma^-$ modes.\\
    Fitting missing masses of $d(K^-, n \pi^-)$ and $d(K^-, n \pi^+)$ \\
    for missing $\Sigma^+$ and $\Sigma^-$ peaks, respectively.\\
    $\Rightarrow$ Event Sample $d(K^-, n \pi^-\pi^+)"n"$ w/ Rejection of $K^0$ and $\Sigma^{\pm}_{forward}$    
  \item Fitting to decompose $d(K^-, n)"n K^0"$ events.
    \begin{enumerate}
    \item Quasi-elastic (1-step) $K^- p \rightarrow K^0 n$
    \item 2-step scattering
    \item direct-$\Lambda(1520)$ production.
    \end{enumerate}
  \end{enumerate}

  Fitting Order:
  $(\textcircled{1.} \rightarrow \textcircled{2.}) \times 5 \mbox{iteration}$
  $\rightarrow \textcircled{3.} \rightarrow \textcircled{2.}$
\end{frame}
