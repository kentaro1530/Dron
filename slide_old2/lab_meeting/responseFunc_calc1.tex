\begin{frame}{Response Function Calculation}
\begin{itemize}
  \item $n_{forward}$ momentum.\\
    $\rightarrow$ Decided from $M_{\pi\Sigma}$ (\& $K^-$ beam momentum). \\
  \item Nucleon momentum in deuteron. \\
    $p_{onshell}+p_{offshell}=p_{d}=(m_d, 0, 0, 0)$ \\
    $p_{onshell}=(m_{p/n}, {\bf p}_{fermi})$ \\
  \item Intermediate $\bar{K}$\\
    Calculated from 1-step $K^- p_{N_1} \rightarrow n_{forward} + \bar{K}$\\
    $p_{\bar{K}}=p_{K^-}+p_{N_1}-n_{forward}$\\
    ($N_1$ is not decide onshell or offshell)
  \end{itemize}
\end{frame}
