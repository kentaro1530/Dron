\begin{frame}{Fitting Function}
  2-step Scattering refer {\scriptsize \href{https://arxiv.org/abs/0804.3479}{L. Lesniak, arXiv:0804.3479v1, 2008.}}
  \begin{align*}
    \scriptsize
    & T^I_2(\bar{K} N, \bar{K}N) = \frac{A^I}{1-iA^{I}k_2+\frac{1}{2}A^{I}R^{I}k_2^2}\\
    & T^I_2(\bar{K} N, \pi \Sigma) =
    \frac{e^{i\delta^{I}}}{\sqrt{k_1}}\frac{\sqrt{{\bf Im}A^{I}-\frac{1}{2}|A^{I}|^2{\bf Im}R^{I}k_2^2}}{1-iA^{I}k_2+\frac{1}{2}A^{I}R^{I}k_2^2}\\
    & T^I_2(\pi \Sigma, \pi \Sigma) =
    \frac{e^{i\delta^{I}}}{k_1}\frac{ \sin\delta^I+i{\bf Im}(e^{-i\delta^I}A^I)k_2-\frac{1}{2}{\bf Im}(e^{-i\delta^I}A^IR^I)k_2^2 }{1-iA^{I}k_2+\frac{1}{2}A^{I}R^{I}k_2^2}\\
%    & \mbox{ \scriptsize (not use) }
%    \frac{e^{i\delta^{I}}}{k_1}\frac{ 1 }{1-iA^{I}k_2+\frac{1}{2}A^{I}R^{I}k_2^2}\\
  \end{align*}
  $A^I$ : Scattering Length\\
  $R^I$ : Effective Range \\
  $\delta^I$ : Phase \\
\end{frame}
