\begin{frame}{Fitting Strategy}
  \begin{itemize}
  \item Make Histogram with Gaussian random.
  \item I fitted histogram in the following three ways.
    \begin{enumerate}
    \item Gaussian (Red)
    \item Gaussian with two $\sigma$ (low/high). Asymmetric (Blue)\\
      \begin{equation*}
        f_{fit}(x)=
        \left\{
        \begin{array}{ll}
          A\exp(\frac{\sigma_h^2}{(x-M)^2}) & (x>M) \\
          A\exp(\frac{\sigma_l^2}{(x-M)^2}) & (x<M)
        \end{array}
        \right.
      \end{equation*}
    \item Mean and RMS of Histogram (Black)
    \end{enumerate}
  \end{itemize}
